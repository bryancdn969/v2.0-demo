map.addFusionTableOverlay()

    The map.addFusionTableOverlay() method imports a kml file from FusionTable. 

    var mapDiv = document.getElementById("map_canvas");
var map = plugin.google.maps.Map.getMap(mapDiv, {
  camera: {
    target: {
      'lat': 51.665173,
      'lng': -3.771648
    },
    zoom: 11
  }
});

map.addFusionTableOverlay({
  'select': 'geometry',
  'from': '1mFQBxJHTlUKIWvM6-ihKvz9vFCBctrmjE7-33Fg',
  'suppressInfoWindows': false
});
