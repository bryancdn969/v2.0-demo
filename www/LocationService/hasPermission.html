LocationService.hasPermission()

    The plugin.google.maps.LocationService.hasPermission() method checks the application has a geolocation permission or not. 

    var button = document.getElementById("execButton");
button.addEventListener("click", function() {
  plugin.google.maps.LocationService.hasPermission(function(hasPermission) {
    if (hasPermission) {
      alert("This app has geolocation permission.");
    } else {
      alert("This app does not have geolocation permission.");
    }
  });
});
