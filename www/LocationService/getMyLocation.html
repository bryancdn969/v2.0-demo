LocationService.getMyLocation()

    The plugin.google.maps.LocationService.getMyLocation() method returns the current device location. You can use this method without creating a map. 

    // Get the current device location "without map"
plugin.google.maps.LocationService.getMyLocation(function(location) {

  // Create a map with the device location
  var mapDiv = document.getElementById('map_canvas');
  var map = plugin.google.maps.Map.getMap(mapDiv, {
    'camera': {
      target: location.latLng,
      zoom: 16
    },
    'controls': {
      'myLocation': true
    }
  });


  // Add a marker
  var text = ["Current your location:\n",
    "latitude:" + location.latLng.lat.toFixed(3),
    "longitude:" + location.latLng.lng.toFixed(3),
    "speed:" + location.speed,
    "time:" + location.time,
    "bearing:" + location.bearing].join("\n");

  var marker = map.addMarker({
    title: text,
    position: location.latLng
  });
  marker.showInfoWindow();

}, function(errorResult) {
  alert(JSON.stringify(errorResult, null, 2));
});
